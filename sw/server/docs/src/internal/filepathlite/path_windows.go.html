// isReservedName reports if name is a Windows reserved device name.
// It does not detect names with an extension, which are also reserved on some Windows versions.
//
// For details, search for PRN in
// https://docs.microsoft.com/en-us/windows/desktop/fileio/naming-a-file.
func ( string) bool {
	// Device names can have arbitrary trailing characters following a dot or colon.
	 := 
	for  := 0;  < len(); ++ {

		case ':', '.':
			 = [:]
		}
	}

		 = [:len()-1]
	}
	if !isReservedBaseName() {
		return false
	}
	if len() == len() {
		return true
	}

	// Some Windows versions consider this a reserved name,
	// while others do not. Use FullPath to see if the name is
	// reserved.

		return true
	}
	return false
}

// volumeNameLen returns length of the leading volume name on Windows.
// It returns 0 elsewhere.
//
// See:
// https://learn.microsoft.com/en-us/dotnet/standard/io/file-path-formats
// https://googleprojectzero.blogspot.com/2016/02/the-definitive-guide-on-win32-to-nt.html
func ( string) int {
	switch {
	case len() >= 2 && [1] == ':':
		// Path starts with a drive letter.
		//
		// Not all Windows functions necessarily enforce the requirement that
		// drive letters be in the set A-Z, and we don't try to here.
		//
		// We don't handle the case of a path starting with a non-ASCII character,
		// in which case the "drive letter" might be multiple bytes long.
		return 2

	case len() == 0 || !IsPathSeparator([0]):
		// Path does not have a volume component.
		return 0

	case pathHasPrefixFold(, `\\.\UNC`):
		// We're going to treat the UNC host and share as part of the volume
		// prefix for historical reasons, but this isn't really principled;
		// Windows's own GetFullPathName will happily remove the first
		// component of the path in this space, converting
		// \\.\unc\a\b\..\c into \\.\unc\a\c.
		return uncLen(, len(`\\.\UNC\`))

	case pathHasPrefixFold(, `\\.`) ||
		pathHasPrefixFold(, `\\?`) || pathHasPrefixFold(, `\??`):
		// Path starts with \\.\, and is a Local Device path; or
		// path starts with \\?\ or \??\ and is a Root Local Device path.
		//
		// We treat the next component after the \\.\ prefix as
		// part of the volume name, which means Clean(`\\?\c:\`)
		// won't remove the trailing \. (See #64028.)
		if len() == 3 {
			return 3 // exactly \\.
		}
		, ,  := cutPath([4:])
		if ! {
			return len()
		}
		return len() - len() - 1

	case len() >= 2 && IsPathSeparator([1]):
		// Path starts with \\, and is a UNC path.
		return uncLen(, 2)
	}
	return 0
}

// postClean adjusts the results of Clean to avoid turning a relative path
// into an absolute or rooted one.
func ( *lazybuf) {
	if .volLen != 0 || .buf == nil {
		return
	}
	// If a ':' appears in the path element at the start of a path,
	// insert a .\ at the beginning to avoid converting relative paths
	// like a/../c: into c:.
	for ,  := range .buf {
		if IsPathSeparator() {
			break
		}
		if  == ':' {
			.prepend('.', Separator)
			return
		}
	}
	// If a path begins with \??\, insert a \. at the beginning
	// to avoid converting paths like \a\..\??\c:\x into \??\c:\x
	// (equivalent to c:\x).
	if len(.buf) >= 3 && IsPathSeparator(.buf[0]) && .buf[1] == '?' && .buf[2] == '?' {
		.prepend(Separator, '.')
	}
}
