func ( *Process) ( Signal) error {
	,  := .handleTransientAcquire()
	switch  {
	case statusDone:
		return ErrProcessDone
	case statusReleased:
		return syscall.EINVAL
	}
	defer .handleTransientRelease()

	if  == Kill {
		var  syscall.Handle
		 := syscall.DuplicateHandle(^syscall.Handle(0), syscall.Handle(), ^syscall.Handle(0), &, syscall.PROCESS_TERMINATE, false, 0)
		if  != nil {
			return NewSyscallError("DuplicateHandle", )
		}
		runtime.KeepAlive()
		defer syscall.CloseHandle()
		 = syscall.TerminateProcess(syscall.Handle(), 1)
		return NewSyscallError("TerminateProcess", )
	}
	// TODO(rsc): Handle Interrupt too?
	return syscall.Errno(syscall.EWINDOWS)
}
