func ( string) ( *dnsConfig) {
	 = &dnsConfig{
		ndots:    1,
		timeout:  5 * time.Second,
		attempts: 2,
	}
	defer func() {
		if len(.servers) == 0 {
			.servers = defaultNS
		}
	}()
	,  := adapterAddresses()
	if  != nil {
		return
	}

	for ,  := range  {
		// Only take interfaces whose OperStatus is IfOperStatusUp(0x01) into DNS configs.
		if .OperStatus != windows.IfOperStatusUp {
			continue
		}

		// Only take interfaces which have at least one gateway
		if .FirstGatewayAddress == nil {
			continue
		}

		for  := .FirstDnsServerAddress;  != nil;  = .Next {
			,  := .Address.Sockaddr.Sockaddr()
			if  != nil {
				continue
			}
			var  IP
			switch sa := .(type) {
			case *syscall.SockaddrInet4:
				 = IPv4(.Addr[0], .Addr[1], .Addr[2], .Addr[3])
			case *syscall.SockaddrInet6:
				 = make(IP, IPv6len)
				copy(, .Addr[:])
				if [0] == 0xfe && [1] == 0xc0 {
					// fec0/10 IPv6 addresses are site local anycast DNS
					// addresses Microsoft sets by default if no other
					// IPv6 DNS address is set. Site local anycast is
					// deprecated since 2004, see
					// https://datatracker.ietf.org/doc/html/rfc3879
					continue
				}
			default:
				// Unexpected type.
				continue
			}
			.servers = append(.servers, JoinHostPort(.String(), "53"))
		}
	}
	return 
}
