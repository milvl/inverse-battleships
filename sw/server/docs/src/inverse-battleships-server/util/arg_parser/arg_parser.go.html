// Arguments struct to hold parsed arguments
type Arguments struct {
	Help          bool
	SocketAddress string
	ConfigPath    string
}

// isFlag checks if an argument is a flag
func ( string) bool {
	return strings.HasPrefix(, "-")
}

// ParseArgs parses command-line arguments
func () (*Arguments, error) {
	if len(os.Args) == 1 {
		return &Arguments{Help: true}, nil
	}

	 := os.Args[1:] // skip the program name
	 := Arguments{
		Help:          false,
		SocketAddress: "",
		ConfigPath:    "",
	}

	// check for help flag
	for  := 0;  < len(); ++ {
		switch [] {
		case "-h", "--help":
			.Help = true
			return &, nil
		}
	}

	// handle other flags
	switch [0] {
	case "-a", "--socket_address":
		if 1 < len() && !isFlag([1]) {
			.SocketAddress = [1]
			return &, nil
		} else {
			return nil, fmt.Errorf("-a/--socket_address requires a value")
		}

	case "-c", "--cfg_path":
		if 1 < len() && !isFlag([1]) {
			.ConfigPath = [1]
			return &, nil
		} else {
			return nil, fmt.Errorf("-c/--cfg_path requires a value")
		}

	default:
		return nil, fmt.Errorf("unknown argument: %s", [0])
	}
}
